//Tính T(n) = 1 x 2 x 3 x … x n

#include <iostream>
using namespace std;

int T(int n) {
    if (n == 1) // điều kiện dừng của đệ quy
        return 1;
    return n * T(n - 1); // gọi lại hàm đệ quy với tham số n - 1
}

int main() {
    int n;
    cout << "Nhap n: ";
    cin >> n;
    cout << "T(" << n << ") = " << T(n) << endl;
    return 0;
}
